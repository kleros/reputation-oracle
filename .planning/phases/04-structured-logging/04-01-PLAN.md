1. **Create `bot/src/logger.ts`** — the central pino logger module (per D-07, D-08).

   The module must:
   - Import `pino` from "pino"
   - Create a default logger writing to **stderr** via `pino.destination(2)` (per D-02, Pitfall 6)
   - Set default level from `process.env.LOG_LEVEL ?? "info"` (per D-03 — logger must work before config.ts loads, per D-08)
   - Configure `redact` paths for top-level properties: `["config.BOT_PRIVATE_KEY", "config.PINATA_JWT", "privateKey", "PINATA_JWT", "BOT_PRIVATE_KEY", "authorization", "Authorization"]` (per D-04)
   - Implement a custom error serializer in `serializers.err` that:
     (a) Calls `pino.stdSerializers.err(err)` for the base serialization
     (b) Recursively walks the serialized object and replaces any string matching `0x[0-9a-fA-F]{64}` with `"[REDACTED_KEY]"` (catches private keys in nested viem error causes — Pitfall 11)
     (c) Replaces any string matching `Bearer [A-Za-z0-9._-]+` with `"Bearer [REDACTED]"` (catches Pinata JWT in HTTP error headers — Pitfall 11)
   - Export the logger instance as the default export and also as a named `logger` export
   - Export a `createChildLogger(module: string)` function that returns `logger.child({ module })`
   - Export a `reconfigureLogLevel(level: string)` function that sets `logger.level = level` — called once from index.ts after config loads (per D-08). This handles the case where LOG_LEVEL in the env differs from default "info" and config.ts validates it.

   Skeleton:
   ```typescript
   import pino from "pino";

   function sanitizeValue(value: string): string {
     return value
       .replace(/0x[0-9a-fA-F]{64}/gi, "[REDACTED_KEY]")
       .replace(/Bearer [A-Za-z0-9._\-]+/g, "Bearer [REDACTED]");
   }

   function sanitizeObject(obj: unknown): unknown {
     if (typeof obj === "string") return sanitizeValue(obj);
     if (Array.isArray(obj)) return obj.map(sanitizeObject);
     if (obj && typeof obj === "object") {
       const result: Record<string, unknown> = {};
       for (const [key, val] of Object.entries(obj)) {
         result[key] = sanitizeObject(val);
       }
       return result;
     }
     return obj;
   }

   export const logger = pino({
     level: process.env.LOG_LEVEL ?? "info",
     serializers: {
       err: (err: unknown) => {
         const serialized = pino.stdSerializers.err(err as Error);
         return sanitizeObject(serialized);
       },
     },
     redact: {
       paths: [
         "config.BOT_PRIVATE_KEY",
         "config.PINATA_JWT",
         "privateKey",
         "PINATA_JWT",
         "BOT_PRIVATE_KEY",
         "authorization",
         "Authorization",
       ],
     },
   }, pino.destination(2));  // fd 2 = stderr

   export function createChildLogger(module: string) {
     return logger.child({ module });
   }

   export function reconfigureLogLevel(level: string): void {
     logger.level = level;
   }

   export default logger;
   ```
